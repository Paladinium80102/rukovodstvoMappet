function main(c)

Главная функция, без неё код не будет работать!
function main(c)
{

}
Код нужно писать обязательно в фигурных скобках.
Командой main мы указываем название нашему исполнителю. В скобках вместо C мы можем указать любой другой символ.
А ТАКЖЕ В MAPPET ВСЕ БУКВЫ, КОТОРЫЕ НАПИСАННЫ ЗАГЛАВНЫМИ ДОЛЖНЫ БЫТЬ ЗАГЛАВНЫМИ, ИНАЧЕ СКРИПТ НЕ БУДЕТ РАБОТАТЬ!!!

c.send()

Команда, отправляющая сообщение в чат.
function main(c)
{
c.send(c)
}
Большинство команд в Mappet'е начинаются с исполнителя функции. Так как мы написали function main(c), то и пишем мы c.send().

var

Команда var задаёт значение переменной
function main(c) {
var a = "Dota 2"
c.send(a)
}

c.getWorld.setBlock() и mappet.createBlockState()

Команда mappet.createBlockState() создаёт данные блока.
Команда c.getWorld.setBlock() использует эти данные, чтобы поставить блок.
function main(c) {
var a = mappet.createBlockState("minecraft:dirt", 1);
c.getWorld.setBlock(a, 8, 5, 8);
}

c.getWorld().clone()

Команда c.getWorld().clone() клонирует блок или область блоков в указанных координатах на другие координаты.
function main(c) {
c.getWorld().clone(8, 5, 8, 3, 5, 3)
}
Чтобы клонировать область блоков:
function main(c) {
c.getWorld().clone(8, 5, 8, 8, 4, 8, 3, 5, 3)
    //координаты области ↑
}

dropitemstack

Эта команда из прошлого примера. Позволяет "выкинуть" предмет на указанные координаты и с указанной скоростью
function main(c) {
var item = mappet.createItem("minecraft:diamond")
c.getWorld().dropItemStack(item, 8, 5, 8)
}
И с указанной скоростью:
function main(c) {
var item = mappet.createItem("minecraft:diamond" )
c.getWorld().dropItemStack(item, 8, 5, 8, 0.5, 0, 0.5)
                                       Скорость ↑
}

random

Команда генерирует рандомное число
От нуля:
function main(c) {
var a = mappet.random(100)
c.send(a)
}
От одного числа до другого:
function main(c) {
var a = mappet.random(0, 100)
c.send(a)
}

if, else if и else

if - это оператор "Если, то". else if - оператор "Иначе если, то". else - оператор "Иначе".
В Mappet эти операторы работают немного иначе, чем в других языках програмирования:
function main(c) {
var a = mappet.random(100);
if (a<=30)
c.send("low")
else if (a<=60, a>=30)
c.send("medium")
else
c.send("high")
}
Условия для этих операторов ставятся в скобках, команда and (и) обозначается знаком "&", а команда "или" (могу ошибаться) знаком (|) - shift+\.

mappet.box()

Создает кубическую область.
function main(c)
{
var subject = c.getSubject();
var subjectPosition = subject.getPosition();
var box = mappet.box(-10, 4, -10, 10, 6, 10);
if (box.contains(subjectPosition)){
c.send("the player in in the box")
}
}
contains - проверяет находятся ли указанные координаты в области.

sendTitle и sendSubtitle

Команда sendTitle() отправляет игроку заголовок, а sendSubtutle() отправляет субтитры, под заголовком.
function main(c)
{
c.getSubject().sendTitle("Hello")
c.getSubject().sendSubtitle("World")
}

raytrace

Эта команда понимает куда направлен курсор игрока.
function main(c)
{
var raytrace = c.getPlayer().rayTrace(10)
if (raytrace.isBlock()) {
c.send("Это блок.")
}

else if (raytrace.isMissed()) {
c.send("Ты промазал!")
}
}

createMorph и setMorph

Команда mappet.createMorph() создаёт морф, который мы можем использовать, например в setMorph().
c.getSubject().setMorph - команда, которая позволяет установить морф для сущности.
function main(c)
{
var morph = mappet.createMorph("Вставьте сюда имя морфа")

c.getSubject().setMorph(morph)
}

который находится слева и выбрать глобальные триггеры или забиндить клавишу, на которую скрипт будет активироваться.

Мы переходим в следующий раздел изучения Mappet'а - интерфейс. Мы будем изучать сразу несколько команд одновременно.

createUI

mappet.createUI() - Очень интересная команда, позволяет создать интерфейс.
function main(c) {
var ui = mappet.createUI()
c.getSubject().openUI(ui);
}
c.getSubject - вторичная сущность, на которую будет действовать следующая команда, можно заменить на getPlayer. openUI - открыть интерфейс.

background

Позволяет установить фон. Его цвет можно менять.
function main(c) {
var ui = mappet.createUI().background()
c.getPlayer().openUI(ui);
}

label

Создаёт строку текста. Можно указать фон и его цвет.
function main(c) {
var ui = mappet.createUI().background()
var a = ui.label("Кубезумие").background(0x99909900)
a.xy(460, 255)
c.getPlayer().openUI(ui);
}
xy - устанавливаем координаты X и Y.

button

Команда позволяет создать кнопку. Пока что она ничего не делает.
function main(c)
{
var a = mappet.createUI()
var b = a.label("Жми, жми!").background(0x99999999)
var d = a.button("Сюда!").background(0x22222999)

b.xy(455, 250).wh(100, 20)
d.xy(425, 275).wh(100, 20)
c.getSubject().openUI(a)
}
wh - устанавливает размеры кнопки. Первое число - ширина, второе - высота.

getData

Команда, с помощью которой мы можем обратится к "ячейке памяти" и получить значение из этой "ячейке".позволяет сделать кнопки активными и т.п.
function main(c)
{
var ui = mappet.createUI(c, "handler")
var label = ui.label("Жми, жми!").background(0x88000000)
var button = ui.button("Start...").id("button");

button.xy(435, 275).wh(80, 20)
label.xy(455, 250)
c.getSubject().openUI(ui)
}

function handler(c)
{
var uicontext = c.getSubject().getUIContext()
var data = uicontext.getData()
var pressed = data.getInt("button")

uicontext.get("button").label("You pressed " + pressed)
}
var ui = mappet.createUI(c, "handler") - Указываем функции, к которым будет относится UI.
id("button") - задаём айди кнопке, чтобы быстро к ней обратиться.
function handler(c) создаём ещё одну функцию. handler - обработчик, нужен для обработки данных.
getUIContext() - получаем контекст интерфейса.
getInt - получаем число
get("button") - обращаемся к айди button

textarea

Команда, которая позволяет вводить текст в интерфейсе.
function main(c)
{
var text = c.getSubject().getStates().getString("Текст");
var ui = mappet.createUI(c, "handler").background();
var textarea = ui.textarea(text).id("textarea").tooltip("Ваш текст");
           // Подсказка, открывающаяся при наведении ↑
textarea.updateDelay(500);
//Задержка обновления ↑ (В миллисекундах)
textarea.xy(300, 150).wh(300, 200);
c.getSubject().openUI(ui);
}

function handler(c)
{
var uiContext = c.getSubject().getUIContext();
var data = uiContext.getData();

if (uiContext.getLast() === "textarea")
{
// Восстанавливаем нашу надпись ↓
c.getSubject().getStates().setString("Текст", data.getString("textarea"));
}
}

textbox

Команда, которая позволяет вводить строку текста.
function main(c)
{
var ui = mappet.createUI(c, "handler").background();
var textbox = ui.textbox().id("textbox").tooltip("Введи сообщение");

textbox.rxy(400, 250).wh(160, 20);
c.getSubject().openUI(ui);
}

function handler(c)
{
var uiContext = c.getSubject().getUIContext();
var data = uiContext.getData();

if (uiContext.getLast() === "textbox")
{
c.getSubject().send(c.getPlayer().getName()+ ": " + data.getString("textbox"));
}
}

click

Создаёт область, которая работает также, как и кнопка.
function main(c) {
var ui = mappet.createUI(c, "handler").background()
var click = ui.click().id("click");
var two = ui.click().id("two")
click.xy(300, 150).wh(300, 150)
two.xy(400, 150).wh(300, 150)
c.getSubject().openUI(ui)
}

function handler(c) {
var uicontext = c.getSubject().getUIContext()
var data = uicontext.getData();

if (uicontext.getLast() === "click") {
c.send("sigma")
}
if (uicontext.getLast() === "two") {
c.send("Балбес")
}
}

keybind

Команда, которая назначает действие на клавишу, внутри интерфейса.
function main(c)
{
var ui = mappet.createUI(c, "handler").background();
var label = ui.label("Нажми на В")

label.xy(455, 250).id("label")
ui.getCurrent().keybind(32, "change", "change label")
c.getPlayer().openUI(ui)
}

function handler(c)
{
var uicontext = c.getSubject().getUIContext()
var key = uicontext.getHotkey()

if (key === "change")
{
var data = uicontext.getData()
var pressed = data.getInt("change")

c.send("sigma")
uicontext.get('label').label("Ты нажал!").xy(460, 250)
var block = mappet.createBlockState("minecraft:stone")
c.getWorld().setBlock(block, 8, 4, 8)
}
}

Stack

ui.item() - команда, позволяющая создать ячейку инвентаря в интерфейсе.
function main(c)
{
var ui = mappet.createUI(c, "handler").background()
var diamond = mappet.createItem("minecraft:diamond")
var item = ui.item().id("item")
var stack = ui.item(diamond).id("stack")

stack.xy(460, 270).wh(24, 24)
item.xy(460, 200).wh(24, 24)

c.getPlayer().openUI(ui)
}

function handler(c)
{
var uiContext = c.getSubject().getUIContext();
var data = uiContext.getData();

if (uiContext.getLast() === "item")
{
var item = mappet.createItem(data.getCompound("item"));
var pos = c.getSubject().getPosition();

c.getWorld().dropItemStack(item, pos.x, pos.y + 2, pos.z);
}
}

Mappet: Графика

rect

Раздел, создающий "Графику" - объект интерфейса, несущий декоративную функцию.

rect

Команда, которая рисует квадрат с заданными координатами, шириной, высотой и цветом.
function main(c) {

var ui = mappet.createUI(c, "handler").background()
var firstbackground = ui.graphics().id("firstbackground")
var secondbackground = ui.graphics().id("secondground")

firstbackground.xy(300, 150).wh(300, 150)

secondbackground.xy(400, 150).wh(300, 150)

firstbackground.rect(-50, 50, 150, 150, 0x88000000)
               // ↑ Использование команды rect
secondbackground.rect(130, 50, 150, 150, 0x88992223)

c.getSubject().openUI(ui)
}

hoverOnly

Команда, с помощью которой объект становится видимым только при наведении.
function main(c)
{
var ui = mappet.createUI().background()
var back = ui.graphics().id("back")

back.xy(325, 200)
back.rect(0, 0, 300, 150, 0x88000000).hoverOnly()
c.getSubject().openUI(ui)
}

Mappet: Предметы и блоки


createItem

Приветсвую в следующей теме!
Здесь мы научимся использовать в Mappet'е ванильные предметы, предметы из модов и предметы, созданные вами, а также добавлять им разные свойства.

createItem()

Эта команда выдает данные для переменной block. Данные мы берём у ванильного блока. Также мы можем указать количество и вариант блока (для предмета: насколько поврежден).
function main(c)
{
var block = mappet.createItem("minecraft:stone")

c.getPlayer().giveItem(block)
}

addCanDestroyBlock

Добавляет предмету возможность ломать блок в режиме приключения
function main(c)
{
var block = mappet.createItem("minecraft:diamond_pickaxe", 32, 5)
block.addCanDestroyBlock("minecraft:stone")
c.getPlayer().giveItem(block)
}
